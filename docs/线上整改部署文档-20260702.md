## 4. 发布命令模板

```bash
cd /www/wwwroot/rd.chaimen666.com && bash <<'BASH'
set -e
PM2="/www/server/nodejs/v22.22.3/bin/pm2"
BRANCH="feature/qiwai-ui-experiment"

git stash push -m "server-pre-release-$(date +%Y%m%d%H%M%S)" || true
git fetch origin "$BRANCH"
git checkout "$BRANCH"
git pull --ff-only origin "$BRANCH"

npm run install:all
export BUILD_COMMIT="$(git rev-parse --short HEAD)"
export BUILD_TIME="$(date -u +"%Y-%m-%dT%H:%M:%SZ")"
npm run build

npm --prefix apps/api run migration:show
npm --prefix apps/api run migration:run

STRICT_RELEASE_VERSION=true WEBROOT="/www/wwwroot/rd.chaimen666.com" ADMIN_WEBROOT="/www/wwwroot/rd.chaimen666.com/admin" npm run publish:webroot

cp .env .env.bak.$(date +%Y%m%d%H%M%S)
sed -i "s|^BUILD_COMMIT=.*|BUILD_COMMIT=$BUILD_COMMIT|" .env || echo "BUILD_COMMIT=$BUILD_COMMIT" >> .env
sed -i "s|^BUILD_TIME=.*|BUILD_TIME=$BUILD_TIME|" .env || echo "BUILD_TIME=$BUILD_TIME" >> .env

set -a
. ./.env
set +a
$PM2 restart activity-api --update-env
$PM2 save
nginx -t && nginx -s reload
BASH
```
